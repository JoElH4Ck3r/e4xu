  231.         /**
  232.          * Sorts the XMLList much like the <code>Array.sort()</code> does.
  233.          * 
  234.          * @param    list    XMLList. The XMLList to be sorted.
  235.          * @param    options    Object. May be of the following types:<br/>
  236.          *             String. If you provide a string, the behavior will be the same as 
  237.          *             of <code>Array.sortOn("parameter")</code><br/>
  238.          *             int. If you provide an integer, then the behavior will be the same
  239.          *             as of <code>Array.sortOn(0-31)</code>. You may use Array constants 
  240.          *             for this sorting operation, the same rules apply.<br/>
  241.          *             Function. If you specify the function as an option, this function will 
  242.          *             be called with the following arguments: previous XML node and next XML node.
  243.          *             The function signature should look like this:<br/>
  244.          *             <code>function sortingFunction(a:XML, b:XML):int</code>.
  245.          * 
  246.          * @return    XMLList. The sorted XMLList.
  247.          * 
  248.          * @throws ArgumentError. "options" argument cannot be of type <i>type</i>!
  249.          */
  250.         public static function sort(list:XMLList, options:Object = null):XMLList
  251.         {
  252.             var arr:Array = listToArray(list);
  253.             switch (true)
  254.             {
  255.                 case (options == null):
  256.                     arr.sort();
  257.                     break;
  258.                 case (options is String):
  259.                 case (options is int):
  260.                     arr.sortOn(options);
  261.                     break;
  262.                 case (options is Function):
  263.                     arr.sort(options);
  264.                     break;
  265.                 default:
  266.                     throw new ArgumentError("\"options\" argument cannot be of type " +
  267.                     getQualifiedClassName(options) + "!");
  268.                     break;
  269.             }
  270.             return arrayToList(arr);
  271.         }

  306.         /**
  307.          * Creates a list of N nodes with the optional randomization.<br/>
  308.          * Note that if you supply quantity greater then the number of nodes, 
  309.          * the list will contain as many nodes as there are in the XML. If randomize
  310.          * is set to <code>false</code> the returned nodes are the first children
  311.          * of the provided XML.
  312.          * 
  313.          * @param    xml            XML. The XML to extract nodes from.
  314.          * @param    quantity    uint. How many nodes to extract.
  315.          * @param    randomise    Boolean. Whether or not to randomize the list.
  316.          * @default    <code>false</code>    The nodes are not randomized by default.
  317.          * 
  318.          * @return    XMLList. The list containing some nodes.
  319.          */
  320.         public static function few(xml:XML, quantity:uint, 
  321.                                             randomise:Boolean = false):XMLList
  322.         {
  323.             var nodes:XMLList = xml.*;
  324.             if (nodes.length() < quantity) quantity = nodes.length();
  325.             var list:XMLList;
  326.             var hash:Array = [];
  327.             var node:XML;
  328.             var i:uint;
  329.             do
  330.             {
  331.                 node = randomise ? nodes[(Math.random() * nodes.length()) >> 0] :
  332.                         nodes[i++];
  333.                 if (randomise && hash.indexOf(node.toXMLString()) < 0)
  334.                 {
  335.                     hash.push(node.toXMLString());
  336.                     list ? list += node : list = XMLList(node);
  337.                 }
  338.                 else if(!randomise) list ? list += node : list = XMLList(node);
  339.             }
  340.             while (list.length() < quantity)
  341.             return list;
  342.         }

  354.         {

  357.             if (_hash[xml.name()]) xml.setName(_hash[xml.name()]);
  358.             else xml.setName(generateUID(xml.name()));

  368.          *

  375.         public static function reverseBatchRename(xml:XML, hash:Object):XML
  376.         {
  377.             _hash = hash;
  378.             var oldName:String = xml.name();
  379.             for(var p:String in _hash)
  380.             {
  381.                 if (_hash[p] == oldName)
  382.                 {
  383.                     xml.setName(p);
  384.                     break;
  385.                 }
  386.             }
  387.             xml.*.(reverseBatchRename(valueOf(), _hash));
  388.             return xml;

  391.         /**

  394.          * 
  395.          * @param    list    XMLList. The list to insert items
  396.          * @param    index    int. The position to insert at.
  397.          * @param    ...xmls    One or more objects to be inserted in the list.
  398.          * 
  399.          * @return    XMLList. The modified XMLList.

  401.         public static function insert(list:XMLList, index:int, ...xmls):XMLList
  402.         {
  403.             if (index > list.length() - 1) index = list.length() - 1;
  404.             if (index < 0) index = 0;
  405.             var toInsert:XMLList = XMLList(xmls.shift());
  406.             while (xmls.length) toInsert += xmls.shift();
  407.             toInsert += list[index].copy();
  408.             list[index] = toInsert;
  409.             return list;
  410.         }
